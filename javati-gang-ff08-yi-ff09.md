```java
class Letter {
    char c;
}

public class PassObject {
    static void f(Letter y){
        y.c = "f";
    }

    public static void main(String[] args){
        Letter x = new Letter();
        x.c = "t";
        System.out.println("x.c="+x.c);
        f(x);
        System.out.println("x.c="+x.c);
    }
}
```
